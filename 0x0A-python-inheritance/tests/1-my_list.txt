=======================
How to use 1-my_list.py
=======================

``obj.print_sorted()`` prints out a sorted (ascending sort) list

::

	>>> MyList = __import__('1-my_list').MyList

	>>> mylist = MyList()
	>>> mylist.append(1)
	>>> mylist.append(4)
	>>> mylist.append(2)
	>>> mylist.append(3)
	>>> mylist.append(5)
	>>> print(mylist)
	[1, 4, 2, 3, 5]
	>>> mylist.print_sorted()
	[1, 2, 3, 4, 5]

	>>> mylist = MyList()
	>>> mylist.append(-1)
	>>> mylist.append(-2)
	>>> mylist.append(-8)
	>>> print(mylist)
	[-1, -2, -8]
	>>> mylist.print_sorted()
	[-8, -2, -1]

	>>> mylist = MyList()
	>>> mylist.append(2)
	>>> mylist.append(10)
	>>> mylist.append(2)
	>>> mylist.append(5)
	>>> print(mylist)
	[2, 10, 2, 5]
	>>> mylist.print_sorted()
	[2, 2, 5, 10]

	>>> mylist = MyList()
	>>> mylist.append(2)
	>>> print(mylist)
	[2]
	>>> mylist.print_sorted()
	[2]

	>>> mylist = MyList()
	>>> print(mylist)
	[]
	>>> mylist.print_sorted()
	[]

	>>> mylist = MyList()
	>>> a = [2, 4, 3, 7]
	>>> mylist.extend(a)
	>>> print(mylist)
	[2, 4, 3, 7]
	>>> mylist.print_sorted()
	[2, 3, 4, 7]
